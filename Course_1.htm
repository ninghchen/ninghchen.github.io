    Arithmetic Operators
  • Addition, +
    • # addition.py
a = 1
b = 2

c = a + b

print('a + b = ', c)
  • Subtraction, -
    • # subtraction.py
a = 1
b = 2

print('a - b = ', a-b)
  • Multiplication, *
    • # multiplication.py
a = 1
b = 2

print('a * b = ', a*b)
  • Division, /
    • # division.py
a = 10
b = 3

print('a / b = ', a/b)
  • Floor Division, //
    • # floor_division.py
a = 10
b = 3

print('a // b = ', a//b)
  • Modulus, %
    • Calculate remainder
    • 14 = 3*4 + 2
    • 14 % 3 = 2
    • # modulus.py
a = 14
b = 3

print('a % b = ', a%b)
  • Exponentiation, **
    • # exponentiation.py
a = 10

print('a^2 = ', a**2)
