import dash
from dash import html
 
# create a Dash app
app = dash.Dash(__name__)
 
# create a web layout
def getLayout():
    return html.Div("Hello World!")

app.layout = getLayout()
 
# start server
if __name__ == "__main__":
    app.run_server(port=8080, debug=True) # turn on Dash tools
            
# access app
http://127.0.0.1:8080/
