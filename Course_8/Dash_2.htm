import dash
from dash import html
from dash import dcc
from dash.dependencies import Input, Output

# create a Dash app
app = dash.Dash(__name__)

# create a web layout
def getLayout():
    return html.Div(
        [
            dcc.Input(id='input'),
            html.Br(),
            html.Label(id = 'output')
        ]
    )

app.layout = getLayout()

# create a callback
@app.callback(
    Output('output', 'children'),
    Input('input', 'value')
)
def update_outputs(input_value):
    if input_value:
        return 'Input: '+input_value
    else:
        return 'Waiting for input ...'

# start server
if __name__ == "__main__":
    app.run_server(port=8080, debug=True) # turn on Dash tools
            
# access app
http://127.0.0.1:8080/
